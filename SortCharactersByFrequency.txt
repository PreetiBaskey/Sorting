//approach - 1 - unordered_map, vector<>, sort
//time complexity - O(n + klogk)
//space complexity - O(n + k)
class Solution {
public:
    string frequencySort(string s) {
        unordered_map<char, int> mp;
        string result;

        //store frequency of a character
        for(char &ch: s) {
            mp[ch]++;
        }

        // Transfer to vector for sorting
        vector<pair<char, int>> charCount;
        for(auto &ele: mp) {
            char ch = ele.first;
            int freq = ele.second;
            charCount.push_back({ch, freq});
        }

        auto myComparator = [&](pair<char, int> &c1, pair<char, int> &c2) {
            return c1.second > c2.second;
        };

        //sorting charCount based on char frequency
        sort(charCount.begin(), charCount.end(), myComparator);

        for(int i = 0; i < charCount.size(); i++) {
            char ch = charCount[i].first;
            int freq = charCount[i].second;
            while(freq--) {
                result += ch;
            }
        }

        return result;
    }
};
