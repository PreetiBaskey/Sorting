//approach - 1 - using sort, tricky counting
//time complexity - O(nlogn + n)
//space complexity - O(sorting functionality space)
class Solution {
public:
    int partitionArray(vector<int>& nums, int k) {
        sort(begin(nums), end(nums));
        int n = nums.size();
        int count = 1; //first subsequence
        int minVal = nums[0]; //this is a part of 1st subsequence

        for(int i = 0; i < n; i++) {
            if(nums[i] - minVal > k) {
                count++;
                minVal = nums[i];
            }
        }

        return count;
    }
};
